# The following are cmd snippets for using git in CLI

in .vim/ directory:

# after working on a file, need to manually re-add into repo index:
git add somefile.txt

# to add a new plugin into my repo, EXAMPLE: 
git submodule add http://github.com/tpope/vim-fugitive.git bundle/fugitive

# after adding submodule, mark them for current repo. Required.
git submodule init
git submodule update
git submodule foreach git submodule init
git submodule foreach git submodule update

# take snapshot (i.e. commit files)
git commit -m "some message"  # github requires -m "message"
# send local changes to github(remote, origin), using "master" branch
git push -u origin master

ln -s ~/.vim/.vimrc ~/.vimrc
ln -s ~/.vim/.bashrc ~/.bashrc

#3. go into .vim/ and update submodules
